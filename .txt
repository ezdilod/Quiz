      let divEl = document.createElement("div");
      contentEl.appendChild(divEl);

      //create img tag and apply the src atr
      let image = document.createElement("img");
      image.src = dataArray.url;
      divEl.appendChild(image);
      // img end

      // add question
      let question = document.createElement("p");
      question.textContent = dataArray.question;
      divEl.appendChild(question);
      // question end

      //answer rendered on page
      let answer = document.createElement("p");
      answer.textContent = dataArray.answer;
      divEl.appendChild(answer);

      //create the answer buttons with value
      let choiceArray = dataArray.choice;
      for (let i = 0; i < choiceArray.length; i++) {
        let answer = choiceArray[i];
        answer = document.createElement("input");
        answer.type = "submit";
        answer.value = choiceArray[i];
        answer.textContent = choiceArray[i];
        divEl.appendChild(answer);
      }
    }
